const crypto = require('crypto');

function hashPassword(pwd) {
  const salt = crypto
    .randomBytes(16).toString('hex');
  return crypto.pbkdf2Sync(
    pwd, salt, 1000, 64, 'sha512'
  ).toString('hex');
}
