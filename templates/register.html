class SecurityScanner {
  constructor() {
    this.rules = [];
    this.parser = new ASTParser();
  }

  addRule(rule) {
    this.rules.push(rule);
  }

  scan(sourceCode) {
    const ast = this.parser.parse(
      sourceCode
    );
    return this.rules.flatMap(
      r => r.detect(ast)
    );
  }
}
